 use registro_db;
 
 
CREATE TABLE IF NOT EXISTS personas (
    id               INT AUTO_INCREMENT PRIMARY KEY,
 
    -- Campos de texto normales
    nombre           VARCHAR(120) NOT NULL,
    email            VARCHAR(180) NOT NULL UNIQUE,
 
    -- ▼ ENUM: la BD solo permite UNO de estos valores exactos
    genero           ENUM('Masculino', 'Femenino', 'No binario', 'Prefiero no decir') NOT NULL,
    nivel_academico  ENUM('Primaria', 'Secundaria', 'Técnico', 'Universitario', 'Posgrado') NOT NULL,
 
    -- ▼ SET: la BD acepta UNO O VARIOS valores de la lista, separados por coma
    idiomas          SET('Español', 'Inglés', 'Francés', 'Portugués', 'Alemán', 'Italiano'),
    habilidades      SET('Java', 'Python', 'JavaScript', 'SQL', 'HTML/CSS', 'Spring', 'React')
);
SELECT * FROM personas;
